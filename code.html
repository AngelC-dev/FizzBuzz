// Get the values from the page
// Start our controller function
function getValues() {
  let startValue = 1;
  let endValue = 100;

  // We call generateNumbers
  let numbers = generateNumbers(startValue, endValue);
  // We call displayNumbers
  displayNumbers(numbers);
}

// Generate numbers from start value to end value
// Logic functions
function generateNumbers(startValue, endValue) {
  let numbers = [];

  // We want to get all numbers from start to end
  for (let index = startValue; index <= endValue; index++) {
    let number;

    if (index % 3 === 0 && index % 5 === 0) {
      number = "FizzBuzz";
    } else if (index % 3 === 0) {
      number = "Fizz";
    } else if (index % 5 === 0) {
      number = "Buzz";
    } else {
      number = index;
    }

    numbers.push(number);
  }

  return numbers;
}

// Display the numbers and mark even numbers bold
// Display or view functions
function displayNumbers(numbers) {
  let templateRows = "";

  for (let index = 0; index < numbers.length; index++) {
    let className = "even";
    let number = numbers[index];

    if (typeof number === "number") {
      if (number % 2 === 0) {
        className = "even";
      } else {
        className = "odd";
      }
    }

    templateRows += `${number}`;
  }

  document.getElementById("results").innerHTML = templateRows;
}
Here's a summary of what each function does:
getValues()

This function retrieves the start and end values from the page, validates them as integers, and calls the `generateNumbers()` function with the valid input. If the input is not valid, it displays an alert.

generateNumbers(startValue, endValue)

This function generates an array of numbers from the `startValue` to the `endValue`. It checks each number and replaces it with "Fizz", "Buzz", or "FizzBuzz" based on the multiples of 3 and 5.

displayNumbers(numbers)

This function takes an array of numbers and displays them on the page. It marks even numbers with the CSS class "even" and odd numbers with the class "odd". It also handles the modified numbers ("Fizz", "Buzz", and "FizzBuzz") by displaying them accordingly.

These functions work together to generate and display a sequence of numbers from 1 to 100, with specific replacements for multiples of 3, 5, and both.

Centus and Fizz Buzz are two different programming exercises that involve manipulating and printing a sequence of numbers based on certain conditions. Here are the differences between them: Centus is a coding exercise where the goal is to print numbers from 1 to 100. Fizz Buzz is a classic coding exercise where the goal is to print numbers from 1 to 100, replacing the multiples of 3 with "Fizz", multiples of 5 with "Buzz", and multiples of both 3 and 5 with "FizzBuzz".
